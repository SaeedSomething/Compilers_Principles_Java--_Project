// Second Class: Circle
class Circle extends Shape {
    int radius;

    // Constructor for Circle
    public Circle(int radius) {
        radius = radius;
    }

    // Overriding area method for circle
    @Override
    public int area() {
        return 3 ;
    }
}

// Abstract class: Shape
abstract class Shape {
    private int x;
    int y;

    // Abstract method for area calculation

}
